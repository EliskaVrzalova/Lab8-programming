	display message "Error: number entered cannot be less than 0.";

	totalSeconds = totalSecs(seconds, minutes, hours); //function call

totalSecs(t_seconds, t_minutes, t_hours)
start function

//variables
totalSeconds = 0;
SECONDS = 60;
